Siriusec Helm chart repo

Add the repo

helm repo add siriusec https://charts.releases.siriusec.dev

Install a chart

See our Helm chart guides for HA setups in EKS or GKE.

siriusec-cluster:
      # Install a single node Siriusec cluster and provision a cert using ACME.
      # Set clusterName to a unique hostname, for example siriusec.example.com
      # Set acmeEmail to receive correspondence from Letsencrypt certificate authority.

      helm install ${RELEASE_NAME?} siriusec/siriusec-cluster \
        --create-namespace \
        --namespace ${NAMESPACE?} \
        --set clusterName=${CLUSTER_NAME?} \
        --set acme=true \
        --set acmeEmail=${EMAIL?}

siriusec-kube-agent:
      # Install or upgrade the Siriusec kube agent which can be used to provide remote access to
      # a remote Kubernetes cluster. Can also be used to run Siriusec database or app service agents.
      # Click the link above to see the README.

      helm install ${RELEASE_NAME?} siriusec/siriusec-kube-agent \
        --create-namespace \
        --namespace ${NAMESPACE?} \
        --set proxyAddr=${PROXY_ENDPOINT?} \
        --set authToken=${JOIN_TOKEN?} \
        --set kubeClusterName=${KUBERNETES_CLUSTER_NAME?}
